A powerful UI library for Roblox with beautiful animations and easy-to-use components.

Recommended for most users

Quick Setup

-- Initialize WindUI
local WindUI = loadstring(game:HttpGet("https://raw.githubusercontent.com/windui/windui/main/loader.lua"))()

-- Create a window
local Window = WindUI:CreateWindow({
    Title = "My Script",
    SubTitle = "WindUI Example",
    Size = UDim2.fromOffset(600, 400),
    Position = UDim2.fromOffset(200, 100)
})

-- Create a tab
local Tab = Window:CreateTab({
    Name = "Main",
    Icon = "rbxassetid://1234567890"
})
